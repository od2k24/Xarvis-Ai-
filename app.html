const chat = document.getElementById("chat"); const input = document.getElementById("input"); let messages = []; function addMessage(text, type) { const div = document.createElement("div"); div.className = "msg " + type; div.textContent = text; chat.appendChild(div); chat.scrollTop = chat.scrollHeight; return div; } async function send() { const text = input.value.trim(); if (!text) return; input.value = ""; // store user message messages.push({ role: "user", content: text }); addMessage(text, "user"); // typing indicator const typing = addMessage("Thinking...", "ai"); try { const res = await fetch("/api/chat", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: text, history: messages.slice(0, -1) }) }); const data = await res.json(); if (!res.ok) { typing.textContent = "Error: " + (data.error || "Failed"); return; } // replace typing typing.textContent = ""; // 🔥 STREAM EFFECT (fake streaming for now) typeText(typing, data.reply || "No response"); messages.push({ role: "assistant", content: data.reply }); } catch (err) { typing.textContent = "Server error"; } } /* ✨ typing animation */ function typeText(el, text) { let i = 0; el.textContent = ""; const interval = setInterval(() => { el.textContent += text[i]; i++; chat.scrollTop = chat.scrollHeight; if (i >= text.length) clearInterval(interval); }, 15); } input.addEventListener("keydown", (e) => { if (e.key === "Enter") send(); });
